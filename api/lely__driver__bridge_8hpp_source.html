111  void fetchRPDO(std::shared_ptr<PDOMap> map)
112  {
113  for (int index = 0; index < 256; index++)
114  {
115  for (int subindex = 1; subindex < 9; subindex++)
116  {
117  auto obj = find(0x1600 + index, subindex);
118  if (obj == nullptr)
119  {
120  continue;
121  }
122  uint32_t data;
123  {
124  data = obj->getVal<CO_DEFTYPE_UNSIGNED32>();
125  }
126  uint8_t tmps = (data >> 8) & 0xFF;
127  uint16_t tmpi = (data >> 16) & 0xFFFF;
128  if (tmpi == 0U)
129  {
130  continue;
131  }
132  pdo_mapping mapping;
133  mapping.is_rpdo = true;
134  mapping.is_tpdo = false;
135  (*map)[tmpi][tmps] = mapping;
136  std::cout << "Found rpdo mapped object: index=" << std::hex << (int)tmpi
137  << " subindex=" << (int)tmps << std::endl;
138  }
139  }
140  }

411  template <typename T>
412  std::future<bool> async_sdo_write_typed(uint16_t idx, uint8_t subidx, T value)
413  {
414  std::unique_lock<std::mutex> lck(sdo_mutex);
415  if (running)
416  {
417  sdo_cond.wait(lck);
418  }
419  running = true;
420 
421  auto prom = std::make_shared<std::promise<bool>>();
422  lely::COSub * sub = this->dictionary_->find(idx, subidx);
423  if (sub == nullptr)
424  {
425  std::cout << "async_sdo_write_typed: id=" << (unsigned int)this->get_id() << " index=0x"
426  << std::hex << (unsigned int)idx << " subindex=" << (unsigned int)subidx
427  << " object does not exist" << std::endl;
428  prom->set_value(false);
429  this->running = false;
430  this->sdo_cond.notify_one();
431  return prom->get_future();
432  }
433 
434  this->SubmitWrite(
435  idx, subidx, value,
436  [this, value, prom](uint8_t id, uint16_t idx, uint8_t subidx, ::std::error_code ec) mutable
437  {
438  if (ec)
439  {
440  prom->set_exception(
441  lely::canopen::make_sdo_exception_ptr(id, idx, subidx, ec, "AsyncDownload"));
442  }
443  else
444  {
445  std::scoped_lock<std::mutex> lck(this->dictionary_mutex_);
446  this->dictionary_->setVal<T>(idx, subidx, value);
447  prom->set_value(true);
448  }
449  std::unique_lock<std::mutex> lck(this->sdo_mutex);
450  this->running = false;
451  this->sdo_cond.notify_one();
452  },
453  20ms);
454  return prom->get_future();
455  }

736  template <typename T>
737  const T universal_get_value(uint16_t index, uint8_t subindex)
738  {
739  T value = 0;
740  bool is_tpdo = false;
741  if (this->pdo_map_->find(index) != this->pdo_map_->end())
742  {
743  auto object = this->pdo_map_->at(index);
744  if (object.find(subindex) != object.end())
745  {
746  auto entry = object.at(subindex);
747  is_tpdo = entry.is_tpdo;
748  }
749  }
750  if (!is_tpdo)
751  {
752  if (sync_sdo_read_typed<T>(index, subindex, value, std::chrono::milliseconds(20)))
753  {
754  return value;
755  }
756  }
757 
758  std::scoped_lock<std::mutex> lck(this->dictionary_mutex_);
759  if (typeid(T) == typeid(uint8_t))
760  {
761  value = this->dictionary_->getVal<CO_DEFTYPE_UNSIGNED8>(index, subindex);
762  }
763  if (typeid(T) == typeid(uint16_t))
764  {
765  value = this->dictionary_->getVal<CO_DEFTYPE_UNSIGNED16>(index, subindex);
766  }
767  if (typeid(T) == typeid(uint32_t))
768  {
769  value = this->dictionary_->getVal<CO_DEFTYPE_UNSIGNED32>(index, subindex);
770  }
771  if (typeid(T) == typeid(int8_t))
772  {
773  value = this->dictionary_->getVal<CO_DEFTYPE_INTEGER8>(index, subindex);
774  }
775  if (typeid(T) == typeid(int16_t))
776  {
777  value = this->dictionary_->getVal<CO_DEFTYPE_INTEGER16>(index, subindex);
778  }
779  if (typeid(T) == typeid(int32_t))
780  {
781  value = this->dictionary_->getVal<CO_DEFTYPE_INTEGER32>(index, subindex);
782  }
783 
784  return value;
785  }
