90  class TPDOWriteTask : public ev::CoTask
91  {
92  public:
93  COData data;
94  LelyDriverBridge * driver;
95  std::mutex mtx;
96  explicit TPDOWriteTask(ev_exec_t * exec) : ev::CoTask(exec)
97  {
98  // Lock and load
99  mtx.lock();
100  }
101  void operator()() noexcept
102  {
103  std::scoped_lock<util::BasicLockable> lk(driver->tpdo_event_mutex);
104  switch (data.type_)
105  {
106  case CODataTypes::COData8:
107  driver->tpdo_mapped[data.index_][data.subindex_] = static_cast<uint8_t>(data.data_);
108  break;
109  case CODataTypes::COData16:
110  driver->tpdo_mapped[data.index_][data.subindex_] = static_cast<uint16_t>(data.data_);
111  break;
112  case CODataTypes::COData32:
113  driver->tpdo_mapped[data.index_][data.subindex_] = static_cast<uint32_t>(data.data_);
114  break;
115  default:
116  break;
117  }
118  driver->master.TpdoWriteEvent(driver->id(), data.index_, data.subindex_);
119  // Unlock when done
120  mtx.unlock();
121  }
122  };
